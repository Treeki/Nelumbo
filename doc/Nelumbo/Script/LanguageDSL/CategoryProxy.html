def line(*args, &block)
        if block_given?
                proxy = LineProxy.new
                proxy.instance_exec &block

                spec = proxy.spec
        else
                spec = {}
        end

        args.each do |number|
                if Range === number
                        number.each{|n| add(n, spec)}
                else
                        add(number, spec)
                end
        end
end
