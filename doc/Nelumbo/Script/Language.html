def define_category(category, &block)
        proxy = LanguageDSL::CategoryProxy.new
        proxy.instance_exec &block

        @categories ||= Set.new
        @categories << category

        @line_spec ||= {}
        @line_spec[category] ||= {}
        @line_spec[category].update(proxy.lines)
end
